### Tanach (Hebrew): Isaiah (Yisheyah) Chapter 28

‎<span id="001">1</span> ‏ה֗וֹי עֲטֶ֤רֶת גֵּאוּת֙ שִׁכֹּרֵ֣י אֶפְרַ֔יִם
וְצִ֥יץ נֹבֵ֖ל צְבִ֣י תִפְאַרְתּ֑וֹ אֲשֶׁ֛ר עַל־רֹ֥אשׁ גֵּֽיא־שְׁמָנִ֖ים
הֲל֥וּמֵי יָֽיִן׃

‎<span id="002">2</span> ‏הִנֵּ֨ה חָזָ֤ק וְאַמִּץ֙ לַֽאדֹנָ֔י כְּזֶ֥רֶם
בָּרָ֖ד שַׂ֣עַר קָ֑טֶב כְּ֠זֶרֶם מַ֣יִם כַּבִּירִ֥ים שֹׁטְפִ֛ים
הִנִּ֥יחַ לָאָ֖רֶץ בְּיָֽד׃

‎<span id="003">3</span> ‏בְּרַגְלַ֖יִם תֵּֽרָמַ֑סְנָה עֲטֶ֥רֶת גֵּא֖וּת
שִׁכּוֹרֵ֥י אֶפְרָֽיִם׃

‎<span id="004">4</span> ‏וְֽהָ֨יְתָ֜ה צִיצַ֤ת נֹבֵל֙ צְבִ֣י
תִפְאַרְתּ֔וֹ אֲשֶׁ֥ר עַל־רֹ֖אשׁ גֵּ֣יא שְׁמָנִ֑ים כְּבִכּוּרָהּ֙
בְּטֶ֣רֶם קַ֔יִץ אֲשֶׁ֨ר יִרְאֶ֤ה הָֽרֹאֶה֙ אוֹתָ֔הּ בְּעוֹדָ֥הּ
בְּכַפּ֖וֹ יִבְלָעֶֽנָּה׃ ס

‎<span id="005">5</span> ‏בַּיּ֣וֹם הַה֗וּא יִֽהְיֶה֙ יְהוָ֣ה צְבָא֔וֹת
לַעֲטֶ֣רֶת צְבִ֔י וְלִצְפִירַ֖ת תִּפְאָרָ֑ה לִשְׁאָ֖ר עַמּֽוֹ׃

‎<span id="006">6</span> ‏וּלְר֖וּחַ מִשְׁפָּ֑ט לַיּוֹשֵׁב֙
עַל־הַמִּשְׁפָּ֔ט וְלִ֨גְבוּרָ֔ה מְשִׁיבֵ֥י מִלְחָמָ֖ה שָֽׁעְרָה׃ ס

‎<span id="007">7</span> ‏וְגַם־אֵ֙לֶּה֙ בַּיַּ֣יִן שָׁג֔וּ וּבַשֵּׁכָ֖ר
תָּע֑וּ כֹּהֵ֣ן וְנָבִיא֩ שָׁג֨וּ בַשֵּׁכָ֜ר נִבְלְע֣וּ מִן־הַיַּ֗יִן
תָּעוּ֙ מִן־הַשֵּׁכָ֔ר שָׁגוּ֙ בָּֽרֹאֶ֔ה פָּק֖וּ פְּלִילִיָּֽה׃

‎<span id="008">8</span> ‏כִּ֚י כָּל־שֻׁלְחָנ֔וֹת מָלְא֖וּ קִ֣יא צֹאָ֑ה
בְּלִ֖י מָקֽוֹם׃ ס

‎<span id="009">9</span> ‏אֶת־מִי֙ יוֹרֶ֣ה דֵעָ֔ה וְאֶת־מִ֖י יָבִ֣ין
שְׁמוּעָ֑ה גְּמוּלֵי֙ מֵֽחָלָ֔ב עַתִּיקֵ֖י מִשָּׁדָֽיִם׃

‎<span id="010">10</span> ‏כִּ֣י צַ֤ו לָצָו֙ צַ֣ו לָצָ֔ו קַ֥ו לָקָ֖ו
קַ֣ו לָקָ֑ו זְעֵ֥יר שָׁ֖ם זְעֵ֥יר שָֽׁם׃

‎<span id="011">11</span> ‏כִּ֚י בְּלַעֲגֵ֣י שָׂפָ֔ה וּבְלָשׁ֖וֹן
אַחֶ֑רֶת יְדַבֵּ֖ר אֶל־הָעָ֥ם הַזֶּֽה׃

‎<span id="012">12</span> ‏אֲשֶׁ֣ר׀ אָמַ֣ר אֲלֵיהֶ֗ם זֹ֤את הַמְּנוּחָה֙
הָנִ֣יחוּ לֶֽעָיֵ֔ף וְזֹ֖את הַמַּרְגֵּעָ֑ה וְלֹ֥א אָב֖וּא שְׁמֽוֹעַ׃

‎<span id="013">13</span> ‏וְהָיָ֨ה לָהֶ֜ם דְּבַר־יְהוָ֗ה צַ֣ו לָצָ֞ו
צַ֤ו לָצָו֙ קַ֤ו לָקָו֙ קַ֣ו לָקָ֔ו זְעֵ֥יר שָׁ֖ם זְעֵ֣יר שָׁ֑ם לְמַ֨עַן
יֵלְכ֜וּ וְכָשְׁל֤וּ אָחוֹר֙ וְנִשְׁבָּ֔רוּ וְנוֹקְשׁ֖וּ וְנִלְכָּֽדוּ׃
פ

‎<span id="014">14</span> ‏לָכֵ֛ן שִׁמְע֥וּ דְבַר־יְהוָ֖ה אַנְשֵׁ֣י
לָצ֑וֹן מֹֽשְׁלֵי֙ הָעָ֣ם הַזֶּ֔ה אֲשֶׁ֖ר בִּירוּשָׁלִָֽם׃

‎<span id="015">15</span> ‏כִּ֣י אֲמַרְתֶּ֗ם כָּרַ֤תְנֽוּ בְרִית֙
אֶת־מָ֔וֶת וְעִם־שְׁא֖וֹל עָשִׂ֣ינוּ חֹזֶ֑ה שיט שׁ֣וֹט שׁוֹטֵ֤ף
כִּֽי־עבר יַֽעֲבֹר֙ לֹ֣א יְבוֹאֵ֔נוּ כִּ֣י שַׂ֧מְנוּ כָזָ֛ב מַחְסֵ֖נוּ
וּבַשֶּׁ֥קֶר נִסְתָּֽרְנוּ׃ ס

‎<span id="016">16</span> ‏לָכֵ֗ן כֹּ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה הִנְנִ֛י
יִסַּ֥ד בְּצִיּ֖וֹן אָ֑בֶן אֶ֣בֶן בֹּ֜חַן פִּנַּ֤ת יִקְרַת֙ מוּסָ֣ד
מוּסָּ֔ד הַֽמַּאֲמִ֖ין לֹ֥א יָחִֽישׁ׃

‎<span id="017">17</span> ‏וְשַׂמְתִּ֤י מִשְׁפָּט֙ לְקָ֔ו וּצְדָקָ֖ה
לְמִשְׁקָ֑לֶת וְיָעָ֤ה בָרָד֙ מַחְסֵ֣ה כָזָ֔ב וְסֵ֥תֶר מַ֖יִם
יִשְׁטֹֽפוּ׃

‎<span id="018">18</span> ‏וְכֻפַּ֤ר בְּרִֽיתְכֶם֙ אֶת־מָ֔וֶת
וְחָזוּתְכֶ֥ם אֶת־שְׁא֖וֹל לֹ֣א תָק֑וּם שׁ֤וֹט שׁוֹטֵף֙ כִּ֣י יַֽעֲבֹ֔ר
וִהְיִ֥יתֶם ל֖וֹ לְמִרְמָֽס׃

‎<span id="019">19</span> ‏מִדֵּ֤י עָבְרוֹ֙ יִקַּ֣ח אֶתְכֶ֔ם
כִּֽי־בַבֹּ֧קֶר בַּבֹּ֛קֶר יַעֲבֹ֖ר בַּיּ֣וֹם וּבַלָּ֑יְלָה וְהָיָ֥ה
רַק־זְוָעָ֖ה הָבִ֥ין שְׁמוּעָֽה׃

‎<span id="020">20</span> ‏כִּֽי־קָצַ֥ר הַמַּצָּ֖ע מֵֽהִשְׂתָּרֵ֑עַ
וְהַמַּסֵּכָ֥ה צָ֖רָה כְּהִתְכַּנֵּֽס׃

‎<span id="021">21</span> ‏כִּ֤י כְהַר־פְּרָצִים֙ יָק֣וּם יְהוָ֔ה
כְּעֵ֖מֶק בְּגִבְע֣וֹן יִרְגָּ֑ז לַעֲשׂ֤וֹת מַעֲשֵׂ֙הוּ֙ זָ֣ר
מַעֲשֵׂ֔הוּ וְלַֽעֲבֹד֙ עֲבֹ֣דָת֔וֹ נָכְרִיָּ֖ה עֲבֹדָתֽוֹ׃

‎<span id="022">22</span> ‏וְעַתָּה֙ אַל־תִּתְלוֹצָ֔צוּ פֶּֽן־יֶחְזְק֖וּ
מֽוֹסְרֵיכֶ֑ם כִּֽי־כָלָ֨ה וְנֶחֱרָצָ֜ה שָׁמַ֗עְתִּי מֵאֵ֨ת אֲדֹנָ֧י
יְהוִ֛ה צְבָא֖וֹת עַל־כָּל־הָאָֽרֶץ׃

‎<span id="023">23</span> ‏הַאֲזִ֥ינוּ וְשִׁמְע֖וּ קוֹלִ֑י הַקְשִׁ֥יבוּ
וְשִׁמְע֖וּ אִמְרָתִֽי׃

‎<span id="024">24</span> ‏הֲכֹ֣ל הַיּ֔וֹם יַחֲרֹ֥שׁ הַחֹרֵ֖שׁ לִזְרֹ֑עַ
יְפַתַּ֥ח וִֽישַׂדֵּ֖ד אַדְמָתֽוֹ׃

‎<span id="025">25</span> ‏הֲלוֹא֙ אִם־שִׁוָּ֣ה פָנֶ֔יהָ וְהֵפִ֥יץ
קֶ֖צַח וְכַמֹּ֣ן יִזְרֹ֑ק וְשָׂ֨ם חִטָּ֤ה שׂוֹרָה֙ וּשְׂעֹרָ֣ה נִסְמָ֔ן
וְכֻסֶּ֖מֶת גְּבֻלָתֽוֹ׃

‎<span id="026">26</span> ‏וְיִסְּר֥וֹ לַמִּשְׁפָּ֖ט אֱלֹהָ֥יו
יוֹרֶֽנּוּ׃

‎<span id="027">27</span> ‏כִּ֣י לֹ֤א בֶֽחָרוּץ֙ י֣וּדַשׁ קֶ֔צַח
וְאוֹפַ֣ן עֲגָלָ֔ה עַל־כַּמֹּ֖ן יוּסָּ֑ב כִּ֧י בַמַּטֶּ֛ה יֵחָ֥בֶט
קֶ֖צַח וְכַמֹּ֥ן בַּשָּֽׁבֶט׃

‎<span id="028">28</span> ‏לֶ֣חֶם יוּדָ֔ק כִּ֛י לֹ֥א לָנֶ֖צַח אָד֣וֹשׁ
יְדוּשֶׁ֑נּוּ וְ֠הָמַם גִּלְגַּ֧ל עֶגְלָת֛וֹ וּפָרָשָׁ֖יו
לֹֽא־יְדֻקֶּֽנּוּ׃

‎<span id="029">29</span> ‏גַּם־זֹ֕את מֵעִ֛ם יְהוָ֥ה צְבָא֖וֹת יָצָ֑אָה
הִפְלִ֣יא עֵצָ֔ה הִגְדִּ֖יל תּוּשִׁיָּֽה׃ ס
