72. ‏الجن‎

'ljn

1 ‏قُلْ أُوحِىَ إِلَىَّ أَنَّهُ ٱسْتَمَعَ نَفَرٌۭ مِّنَ ٱلْجِنِّ فَقَالُوٓا۟ إِنَّا سَمِعْنَا قُرْءَانًا عَجَبًۭا ‎

 qul 'ʔuwḥiya 'ʔilay:a 'ʔan:ahu 'stamaʕa nafarunm m:ina 'ljin:i faqa'luwʔ' 'ʔin:a' samiʕna' qurʔa'nan' ʕajabanm'

2 ‏يَهْدِىٓ إِلَى ٱلرُّشْدِ فَـَٔامَنَّا بِهِۦ ۖ وَلَن نُّشْرِكَ بِرَبِّنَآ أَحَدًۭا ‎

 yahdiyʔ 'ʔilay 'lr:uʃdi faʔa'man:a' bihiy sl' walan n:uʃrika birab:ina'ʔ 'ʔaḥadanm'

3 ‏وَأَنَّهُۥ تَعَلَىٰ جَدُّ رَبِّنَا مَا ٱتَّخَذَ صَحِبَةًۭ وَلَا وَلَدًۭا ‎

 wa'ʔan:ahuw taʕalay' jad:u rab:ina' ma' 't:axaða ṣaḥibatanm wala' waladanm'

4 ‏وَأَنَّهُۥ كَانَ يَقُولُ سَفِيهُنَا عَلَى ٱللَّهِ شَطَطًۭا ‎

 wa'ʔan:ahuw ka'na yaquwlu safiyhuna' ʕalay 'll:ahi ʃaṭaṭanm'

5 ‏وَأَنَّا ظَنَنَّآ أَن لَّن تَقُولَ ٱلْإِنسُ وَٱلْجِنُّ عَلَى ٱللَّهِ كَذِبًۭا ‎

 wa'ʔan:a' ẓanan:a'ʔ 'ʔan l:an taquwla 'l'ʔinsu wa'ljin:u ʕalay 'll:ahi kaðibanm'

6 ‏وَأَنَّهُۥ كَانَ رِجَالٌۭ مِّنَ ٱلْإِنسِ يَعُوذُونَ بِرِجَالٍۢ مِّنَ ٱلْجِنِّ فَزَادُوهُمْ رَهَقًۭا ‎

 wa'ʔan:ahuw ka'na rija'lunm m:ina 'l'ʔinsi yaʕuwðuwna birija'linm m:ina 'ljin:i faza'duwhum rahaqanm'

7 ‏وَأَنَّهُمْ ظَنُّوا۟ كَمَا ظَنَنتُمْ أَن لَّن يَبْعَثَ ٱللَّهُ أَحَدًۭا ‎

 wa'ʔan:ahum ẓan:uw' kama' ẓanantum 'ʔan l:an yabʕaθa 'll:ahu 'ʔaḥadanm'

8 ‏وَأَنَّا لَمَسْنَا ٱلسَّمَآءَ فَوَجَدْنَهَا مُلِئَتْ حَرَسًۭا شَدِيدًۭا وَشُهُبًۭا ‎

 wa'ʔan:a' lamasna' 'ls:ama'ʔʔa fawajadnaha' muliyʔat ḥarasanm' ʃadiydanm' waʃuhubanm'

9 ‏وَأَنَّا كُنَّا نَقْعُدُ مِنْهَا مَقَعِدَ لِلسَّمْعِ ۖ فَمَن يَسْتَمِعِ ٱلْءَانَ يَجِدْ لَهُۥ شِهَابًۭا رَّصَدًۭا ‎

 wa'ʔan:a' kun:a' naqʕudu minha' maqaʕida lils:amʕi sl' faman yastamiʕi 'lʔa'na yajid lahuw ʃiha'banm' r:aṣadanm'

10 ‏وَأَنَّا لَا نَدْرِىٓ أَشَرٌّ أُرِيدَ بِمَن فِى ٱلْأَرْضِ أَمْ أَرَادَ بِهِمْ رَبُّهُمْ رَشَدًۭا ‎

 wa'ʔan:a' la' nadriyʔ 'ʔaʃar:un 'ʔuriyda biman fiy 'l'ʔarḍi 'ʔam 'ʔara'da bihim rab:uhum raʃadanm'

11 ‏وَأَنَّا مِنَّا ٱلصَّلِحُونَ وَمِنَّا دُونَ ذَلِكَ ۖ كُنَّا طَرَآئِقَ قِدَدًۭا ‎

 wa'ʔan:a' min:a' 'lṣ:aliḥuwna wamin:a' duwna ðalika sl' kun:a' ṭara'ʔyʔiqa qidadanm'

12 ‏وَأَنَّا ظَنَنَّآ أَن لَّن نُّعْجِزَ ٱللَّهَ فِى ٱلْأَرْضِ وَلَن نُّعْجِزَهُۥ هَرَبًۭا ‎

 wa'ʔan:a' ẓanan:a'ʔ 'ʔan l:an n:uʕjiza 'll:aha fiy 'l'ʔarḍi walan n:uʕjizahuw harabanm'

13 ‏وَأَنَّا لَمَّا سَمِعْنَا ٱلْهُدَىٰٓ ءَامَنَّا بِهِۦ ۖ فَمَن يُؤْمِنۢ بِرَبِّهِۦ فَلَا يَخَافُ بَخْسًۭا وَلَا رَهَقًۭا ‎

 wa'ʔan:a' lam:a' samiʕna' 'lhuday'ʔ ʔa'man:a' bihiy sl' faman yuwʔminm birab:ihiy fala' yaxa'fu baxsanm' wala' rahaqanm'

14 ‏وَأَنَّا مِنَّا ٱلْمُسْلِمُونَ وَمِنَّا ٱلْقَسِطُونَ ۖ فَمَنْ أَسْلَمَ فَأُو۟لَٓئِكَ تَحَرَّوْا۟ رَشَدًۭا ‎

 wa'ʔan:a' min:a' 'lmuslimuwna wamin:a' 'lqasiṭuwna sl' faman 'ʔaslama fa'ʔuwlaʔyʔika taḥar:aw' raʃadanm'

15 ‏وَأَمَّا ٱلْقَسِطُونَ فَكَانُوا۟ لِجَهَنَّمَ حَطَبًۭا ‎

 wa'ʔam:a' 'lqasiṭuwna faka'nuw' lijahan:ama ḥaṭabanm'

16 ‏وَأَلَّوِ ٱسْتَقَمُوا۟ عَلَى ٱلطَّرِيقَةِ لَأَسْقَيْنَهُم مَّآءً غَدَقًۭا ‎

 wa'ʔal:awi 'staqamuw' ʕalay 'lṭ:ariyqati la'ʔasqaynahum m:a'ʔʔan ɣadaqanm'

17 ‏لِّنَفْتِنَهُمْ فِيهِ ۚ وَمَن يُعْرِضْ عَن ذِكْرِ رَبِّهِۦ يَسْلُكْهُ عَذَابًۭا صَعَدًۭا ‎

 l:inaftinahum fiyhi j waman yuʕriḍ ʕan ðikri rab:ihiy yaslukhu ʕaða'banm' ṣaʕadanm'

18 ‏وَأَنَّ ٱلْمَسَجِدَ لِلَّهِ فَلَا تَدْعُوا۟ مَعَ ٱللَّهِ أَحَدًۭا ‎

 wa'ʔan:a 'lmasajida lil:ahi fala' tadʕuw' maʕa 'll:ahi 'ʔaḥadanm'

19 ‏وَأَنَّهُۥ لَمَّا قَامَ عَبْدُ ٱللَّهِ يَدْعُوهُ كَادُوا۟ يَكُونُونَ عَلَيْهِ لِبَدًۭا ‎

 wa'ʔan:ahuw lam:a' qa'ma ʕabdu 'll:ahi yadʕuwhu ka'duw' yakuwnuwna ʕalayhi libadanm'

20 ‏قُلْ إِنَّمَآ أَدْعُوا۟ رَبِّى وَلَآ أُشْرِكُ بِهِۦٓ أَحَدًۭا ‎

 qul 'ʔin:ama'ʔ 'ʔadʕuw' rab:iy wala'ʔ 'ʔuʃriku bihiyʔ 'ʔaḥadanm'

21 ‏قُلْ إِنِّى لَآ أَمْلِكُ لَكُمْ ضَرًّۭا وَلَا رَشَدًۭا ‎

 qul 'ʔin:iy la'ʔ 'ʔamliku lakum ḍar:anm' wala' raʃadanm'

22 ‏قُلْ إِنِّى لَن يُجِيرَنِى مِنَ ٱللَّهِ أَحَدٌۭ وَلَنْ أَجِدَ مِن دُونِهِۦ مُلْتَحَدًا ‎

 qul 'ʔin:iy lan yujiyraniy mina 'll:ahi 'ʔaḥadunm walan 'ʔajida min duwnihiy multaḥadan'

23 ‏إِلَّا بَلَغًۭا مِّنَ ٱللَّهِ وَرِسَلَتِهِۦ ۚ وَمَن يَعْصِ ٱللَّهَ وَرَسُولَهُۥ فَإِنَّ لَهُۥ نَارَ جَهَنَّمَ خَلِدِينَ فِيهَآ أَبَدًا ‎

 'ʔil:a' balaɣanm' m:ina 'll:ahi warisalatihiy j waman yaʕṣi 'll:aha warasuwlahuw fa'ʔin:a lahuw na'ra jahan:ama xalidiyna fiyha'ʔ 'ʔabadan'

24 ‏حَتَّىٰٓ إِذَا رَأَوْا۟ مَا يُوعَدُونَ فَسَيَعْلَمُونَ مَنْ أَضْعَفُ نَاصِرًۭا وَأَقَلُّ عَدَدًۭا ‎

 ḥat:ay'ʔ 'ʔiða' ra'ʔaw' ma' yuwʕaduwna fasayaʕlamuwna man 'ʔaḍʕafu na'ṣiranm' wa'ʔaqal:u ʕadadanm'

25 ‏قُلْ إِنْ أَدْرِىٓ أَقَرِيبٌۭ مَّا تُوعَدُونَ أَمْ يَجْعَلُ لَهُۥ رَبِّىٓ أَمَدًا ‎

 qul 'ʔin 'ʔadriyʔ 'ʔaqariybunm m:a' tuwʕaduwna 'ʔam yajʕalu lahuw rab:iyʔ 'ʔamadan'

26 ‏عَلِمُ ٱلْغَيْبِ فَلَا يُظْهِرُ عَلَىٰ غَيْبِهِۦٓ أَحَدًا ‎

 ʕalimu 'lɣaybi fala' yuẓhiru ʕalay' ɣaybihiyʔ 'ʔaḥadan'

27 ‏إِلَّا مَنِ ٱرْتَضَىٰ مِن رَّسُولٍۢ فَإِنَّهُۥ يَسْلُكُ مِنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦ رَصَدًۭا ‎

 'ʔil:a' mani 'rtaḍay' min r:asuwlinm fa'ʔin:ahuw yasluku minm bayni yadayhi wamin xalfihiy raṣadanm'

28 ‏لِّيَعْلَمَ أَن قَدْ أَبْلَغُوا۟ رِسَلَتِ رَبِّهِمْ وَأَحَاطَ بِمَا لَدَيْهِمْ وَأَحْصَىٰ كُلَّ شَىْءٍ عَدَدًۢا ‎

 l:iyaʕlama 'ʔan qad 'ʔablaɣuw' risalati rab:ihim wa'ʔaḥa'ṭa bima' ladayhim wa'ʔaḥṣay' kul:a ʃayʔin ʕadadanm'
